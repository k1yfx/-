#include<stdio.h>
void main()
{
	int a=0,b=0,c=0;
	char n=0,m=0;
	scanf("%d%c%d%c%d",&a,&n,&b,&m,&c);
	if('+'==n)
		if(a+b==c)
			printf("Perfect");
		else
			{
				printf("Error\n");
				printf("%d+%d=%d",a,b,a+b);
		    }
	else if('-'==n)
		if(a-b==c)
			printf("Perfect");
		else
		{
			printf("Error\n");
			printf("%d-%d=%d",a,b,a-b);
		}
	
}

#####实验二########
#include<stdio.h>
void main()
{
int a,b;
char op;
scanf("%d%c%d",&a,&op,&b);
switch(op)
	{
	case '+': printf("%d+%d=%d",a,b,a+b);
		break;
	case '-': printf("%d-%d=%d",a,b,a-b);
		break;
	case '*':printf("%d*%d=%d",a,b,a*b);
		break;
	case '/':printf("%d/%d=%d",a,b,a/b);
		break;
	default:printf("输入错误")；
	}
}
