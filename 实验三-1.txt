#include <stdio.h>
void main( )
{
	int a[3][3], i=0, j=0, k, flag, max, col;
	/**********found**********/
	for (i = 0; i <= 2; i++)                 //输入数组
	{
		/**********found**********/
		for(j = 0; j <= 2; j++)
		{
		scanf("%d", &a[i][j]);
		}
	}
	printf("\n");
	for (i = 0; i < 3; i++)
		{
		max = a[i][0];
		col = 0;
		for (j = 1; j < 3; j++)
		{
			if (a[i][j]>max)		//第i行最大值max，列号col
			{
				max = a[i][j];
				/**********found**********/
				col = j;
			}
		}
		flag = 1;
		for (k = 0; k < 3; k++)
		{
			/**********found**********/
			if (max > a[k][col])	 //max大于[k][col]置0
			{
				flag = 0;
				break;
			}
		}
		/**********found**********/
		if (flag == 1)
		{
			printf("row=%d, column=%d, value=%d\n", i, col, max);
		}
	}
}

#include <stdio.h>
#include<string.h>
void main( )
{
	char arr[201],a=0,b=0,c=0,d=0,e=0,i=0;
	gets(arr);
	for(i=0;i<strlen(arr);i++)
	{
		if(arr[i]<0)	//中文统计
		{
			a++;
			i++;
			continue;
		}
		if(arr[i]>='A'&&arr[i]<='z')	//英文统计
		{
			b++;
			continue;
		}
		if(arr[i]==' ')	//空格统计	
		{
			c++;
			continue;
		}
		if(arr[i]>='0'&&arr[i]<='9')	//数字字符
		{
			d++;
			continue;
		}
			//其它字符统计
		e++;
	}
	printf("中文字符：%d  英文字符：%d  空格字符：%d  数字字符：%d  其它字符：%d",a,b,c,d,e);
}

#include<stdio.h>
#include<stdlib.h>
#define N 5
void main()
{
	int a[N][N], i, j, k, t;
	printf(" THE ARRAY\n");
	for ( i =0; i<N; i++ )
	{
		for ( j =0; j<N; j++ )
		/************found************/
		{
				a[i][j] = rand()%30;
			printf( "%4d", a[i][j] );
		}
		printf("\n");
	}
	i = 0;
	/************found************/
	j = N-1;
	while ( i < j )
	{
		for (k=0; k<N; k++)
		/************found************/
		{ t=a[i][k];a[i][k]=a[j][k]; a[j][k] = t; }
		i++;
		/************found************/
		j--;
	}
	printf ("\n THE RESULT \n");
	for ( i =0; i<N; i++ )
		{
		for ( j =0; j<N; j++ )
		printf( "%4d", a[i][j] );
		printf("\n");
		}
}

		a[i][0]=i+1;

	}

	{
		for(j=0;j<=b-1;j++)
